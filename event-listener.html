    <script>
        document.onkeydown = checkKey;
        function checkKey(e) {
            e = e || window.event;
            let svg = document.getElementsByTagName("svg")[0];
            if (e.keyCode == '39') {
                svg.innerHTML = '<circle r="45" cx="50" cy="50" fill="red" />';
            }
            else if(e.keyCode == '37'){
                svg.innerHTML = '<rect width="200" height="100" x="10" y="10" rx="20" ry="20" fill="blue" />';
            }
    
        }
    </script>
